fn main() {
    let n: int = 5;
    let sonuc: int = 1;
    
    // Faktoriyel Hesaplama
    while n > 0 {
        sonuc = sonuc * n;
        n = n - 1;
    }
    
    print_str("5 Faktoriyel Sonucu:");
    print(sonuc);
}
